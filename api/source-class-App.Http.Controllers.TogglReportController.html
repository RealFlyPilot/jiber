 48:     if (!$request->date)

 68:     $report->save();
 69: 
 70:         // Get current user from Toggl, so we can filter time entries

 74:     $args = array(
 75:       'user_agent'   => 'thaissa.mendes@gmail.com',
 76:       'workspace_id' => (int)$request->workspace,
 77:       'user_ids'     => $current_user['id'],

 79:
